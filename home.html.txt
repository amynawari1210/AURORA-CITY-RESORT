  <script>
    const $ = (s)=>document.querySelector(s);

    function toast(title, msg){
      $("#toastTitle").textContent = title;
      $("#toastMsg").textContent = msg;
      const t = $("#toast");
      t.style.display = "block";
      t.style.opacity = "1";
      clearTimeout(window.__toastT);
      window.__toastT = setTimeout(()=>{ t.style.opacity = "0"; }, 2600);
      window.__toastT2 = setTimeout(()=>{ t.style.display = "none"; }, 3100);
    }

    // Set default dates: tomorrow + 2 nights
    (function initDates(){
      const now = new Date();
      const checkin = new Date(now.getFullYear(), now.getMonth(), now.getDate()+1);
      const checkout = new Date(now.getFullYear(), now.getMonth(), now.getDate()+3);
      const toISO = (d)=> d.toISOString().slice(0,10);
      $("#checkin").value = toISO(checkin);
      $("#checkout").value = toISO(checkout);
      $("#year").textContent = now.getFullYear();
    })();

    const basePrices = {
      "Deluxe City View": 680,
      "Premier Sea View": 820,
      "Executive Suite": 1180,
      "Family Suite": 1380,
      "Signature Penthouse": 2680
    };

    function updatePrice(){
      const room = $("#room").value;
      const guests = parseInt($("#guests").value, 10) || 2;
      const p = (basePrices[room] || 680) + (guests>2 ? (guests-2)*90 : 0);
      $("#price").textContent = `RM ${p.toLocaleString("en-MY")}`;
    }
    $("#room").addEventListener("change", updatePrice);
    $("#guests").addEventListener("change", updatePrice);
    updatePrice();

    // Mock availability search
    $("#mockSearch").addEventListener("click", ()=>{
      const inD = $("#checkin").value;
      const outD = $("#checkout").value;
      if(!inD || !outD){
        toast("Missing info", "Please select check-in and check-out dates.");
        return;
      }
      const a = new Date(inD), b = new Date(outD);
      if(b <= a){
        toast("Date issue", "Check-out must be after check-in.");
        return;
      }

      // Deterministic “availability” by day number (so it looks consistent on set)
      const seed = a.getDate() + a.getMonth()*31 + a.getFullYear();
      const available = (seed % 5) !== 0; // 1 out of 5 days: limited
      const status = available ? "Available" : "Limited";
      $("#liveStatus").textContent = status;
      $("#liveStatus").style.color = available ? "rgba(76,227,138,.95)" : "rgba(255,184,107,.95)";

      toast("Availability (Mock)", available
        ? "Rooms available. Proceed to booking (mock)."
        : "Limited availability. Try different dates (mock)."
      );
    });

    // Copy email
    $("#copyEmail").addEventListener("click", async ()=>{
      const email = "concierge@auroracityresort.example";
      try{
        await navigator.clipboard.writeText(email);
        toast("Copied", "Email copied to clipboard.");
      }catch(e){
        toast("Copy failed", "Clipboard not available on this browser.");
      }
    });

    // Mock enquiry
    $("#openEnquiry").addEventListener("click", ()=>{
      toast("Enquiry sent (Mock)", "Thank you. Our concierge will respond shortly.");
    });

      toast("Menu", "Scroll to sections: Rooms, Dining, Facilities, Gallery, Events, Contact.");
    });
